    public static void playBeepSong() {
    int[] notes = { 523, 587, 659, 698, 784, 880, 988, 1047 }; // Frequencies in Hz
    
    int time = 3500; // Duration of each note 
    for (int note : notes) {
        
        try {
            Thread.sleep(time); // Pause for the duration of the note
        
        } catch (InterruptedException e) {
        
            e.printStackTrace();
        }
        System.out.print("\u0007"); // Play the note by printing the ASCII bell character
        try {
            Thread.sleep(50); // Pause briefly between notes
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}

    void DispCrd(int x, int y, int cardNumber)

    {
        for (int i=0; i < 10; i++)
        {
            for (int j=0; j < 15; j++)
            {
                if (array[cardNumber][i][j] == 0)

                    
                printxy(x+j,y+i,fcolor[cardNumber][i][j] + bcolor[cardNumber][i][j] + " ");


                else if (array[cardNumber][i][j] >0 && array[cardNumber][i][j] < 10)

                
                printxy(x+j,y+i,fcolor[cardNumber][i][j] + bcolor[cardNumber][i][j] + String.valueOf(array[cardNumber][i][j]));

                else if(array[cardNumber][i][j] == 10){

                    printxy(x+j-1,y+i,fcolor[cardNumber][i][j] + bcolor[cardNumber][i][j] + "1");

                    printxy(x+j,y+i,fcolor[cardNumber][i][j] + bcolor[cardNumber][i][j] + "0");

                }


                else
                    printxy(x+j,y+i,fcolor[cardNumber][i][j] + bcolor[cardNumber][i][j] +Symbols.EXTENDED[array[cardNumber][i][j]]);

            }

            System.out.println(" ");

        }

    }
